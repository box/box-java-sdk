062     * Adds a collaborator to this folder.
063     * @param  collaborator the collaborator to add.
064     * @param  role         the role of the collaborator.
065     * @return              info about the new collaboration.
066     */
067    public BoxCollaboration.Info collaborate(BoxCollaborator collaborator, BoxCollaboration.Role role) {
068        JsonObject accessibleByField = new JsonObject();
069        accessibleByField.add("id", collaborator.getID());
070
071        if (collaborator instanceof BoxUser) {
072            accessibleByField.add("type", "user");
073        } else if (collaborator instanceof BoxGroup) {
074            accessibleByField.add("type", "group");
075        } else {
076            throw new IllegalArgumentException("The given collaborator is of an unknown type.");
077        }
078
079        return this.collaborate(accessibleByField, role);
080    }

097    private BoxCollaboration.Info collaborate(JsonObject accessibleByField, BoxCollaboration.Role role) {
098        BoxAPIConnection api = this.getAPI();
099        URL url = ADD_COLLABORATION_URL.build(api.getBaseURL());
100
101        JsonObject itemField = new JsonObject();
102        itemField.add("id", this.getID());
103        itemField.add("type", "folder");
104
105        JsonObject requestJSON = new JsonObject();
106        requestJSON.add("item", itemField);
107        requestJSON.add("accessible_by", accessibleByField);
108        requestJSON.add("role", role.toJSONString());
109
110        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
111        request.setBody(requestJSON.toString());
112        BoxJSONResponse response = (BoxJSONResponse) request.send();
113        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
114
115        BoxCollaboration newCollaboration = new BoxCollaboration(api, responseJSON.get("id").asString());
116        BoxCollaboration.Info info = newCollaboration.new Info(responseJSON);
117        return info;
118    }

132    /**
133     * Gets information about all of the collaborations for this folder.
134     * @return a collection of information about the collaborations for this folder.
135     */
136    public Collection<BoxCollaboration.Info> getCollaborations() {
137        BoxAPIConnection api = this.getAPI();
138        URL url = GET_COLLABORATIONS_URL.build(api.getBaseURL(), this.getID());
139
140        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
141        BoxJSONResponse response = (BoxJSONResponse) request.send();
142        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
143
144        int entriesCount = responseJSON.get("total_count").asInt();
145        Collection<BoxCollaboration.Info> collaborations = new ArrayList<BoxCollaboration.Info>(entriesCount);
146        JsonArray entries = responseJSON.get("entries").asArray();
147        for (JsonValue entry : entries) {
148            JsonObject entryObject = entry.asObject();
149            BoxCollaboration collaboration = new BoxCollaboration(api, entryObject.get("id").asString());
150            BoxCollaboration.Info info = collaboration.new Info(entryObject);
151            collaborations.add(info);
152        }
153
154        return collaborations;
155    }

343    /**
344     * Uploads a new file to this folder with custom upload parameters.
345     * @param  uploadParams the custom upload parameters.
346     * @return              the uploaded file's info.
347     */
348    public BoxFile.Info uploadFile(FileUploadParams uploadParams) {
349        URL uploadURL = UPLOAD_FILE_URL.build(this.getAPI().getBaseUploadURL());
350        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
351
352        JsonObject fieldJSON = new JsonObject();
353        JsonObject parentIdJSON = new JsonObject();
354        parentIdJSON.add("id", getID());
355        fieldJSON.add("name", uploadParams.getName());
356        fieldJSON.add("parent", parentIdJSON);
357
358        if (uploadParams.getCreated() != null) {
359            fieldJSON.add("content_created_at", BoxDateFormat.format(uploadParams.getCreated()));
360        }
361
362        if (uploadParams.getModified() != null) {
363            fieldJSON.add("content_modified_at", BoxDateFormat.format(uploadParams.getModified()));
364        }
365
366        request.putField("attributes", fieldJSON.toString());
367
368        if (uploadParams.getSize() > 0) {
369            request.setFile(uploadParams.getContent(), uploadParams.getName(), uploadParams.getSize());
370        } else {
371            request.setFile(uploadParams.getContent(), uploadParams.getName());
372        }
373
374        BoxJSONResponse response;
375        if (uploadParams.getProgressListener() == null) {
376            response = (BoxJSONResponse) request.send();
377        } else {
378            response = (BoxJSONResponse) request.send(uploadParams.getProgressListener());
379        }
380        JsonObject collection = JsonObject.readFrom(response.getJSON());
381        JsonArray entries = collection.get("entries").asArray();
382        JsonObject fileInfoJSON = entries.get(0).asObject();
383        String uploadedFileID = fileInfoJSON.get("id").asString();
384
385        BoxFile uploadedFile = new BoxFile(getAPI(), uploadedFileID);
386        return uploadedFile.new Info(fileInfoJSON);
387    }

415    /**
416     * Retrieves a specific range of child items in this folder.
417     * @param  offset the index of the first child item to retrieve.
418     * @param  limit  the maximum number of children to retrieve after the offset.
419     * @param  fields the fields to retrieve.
420     * @return        a partial collection containing the specified range of child items.
421     */
422    public PartialCollection<BoxItem.Info> getChildrenRange(long offset, long limit, String... fields) {
423        QueryStringBuilder builder = new QueryStringBuilder()
424            .appendParam("limit", limit)
425            .appendParam("offset", offset);
426
427        if (fields.length > 0) {
428            builder.appendParam("fields", fields).toString();
429        }
430
431        URL url = GET_ITEMS_URL.buildWithQuery(getAPI().getBaseURL(), builder.toString(), getID());
432        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
433        BoxJSONResponse response = (BoxJSONResponse) request.send();
434        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
435
436        String totalCountString = responseJSON.get("total_count").toString();
437        long fullSize = Double.valueOf(totalCountString).longValue();
438        PartialCollection<BoxItem.Info> children = new PartialCollection<BoxItem.Info>(offset, limit, fullSize);
439        JsonArray jsonArray = responseJSON.get("entries").asArray();
440        for (JsonValue value : jsonArray) {
441            JsonObject jsonObject = value.asObject();
442            BoxItem.Info parsedItemInfo = (BoxItem.Info) BoxResource.parseInfo(this.getAPI(), jsonObject);
443            if (parsedItemInfo != null) {
444                children.add(parsedItemInfo);
445            }
446        }
447        return children;
448    }

481    /**
482     * Contains information about a BoxFolder.
483     */
484    public class Info extends BoxItem.Info {
485        private BoxUploadEmail uploadEmail;
486        private boolean hasCollaborations;
487        private SyncState syncState;
488        private EnumSet<Permission> permissions;
489        private boolean canNonOwnersInvite;
490
491        /**
492         * Constructs an empty Info object.
493         */
494        public Info() {
495            super();
496        }
497
498        /**
499         * Constructs an Info object by parsing information from a JSON string.
500         * @param  json the JSON string to parse.
501         */
502        public Info(String json) {
503            super(json);
504        }
505
506        /**
507         * Constructs an Info object using an already parsed JSON object.
508         * @param  jsonObject the parsed JSON object.
509         */
510        Info(JsonObject jsonObject) {
511            super(jsonObject);
512        }
513
514        /**
515         * Gets the upload email for the folder.
516         * @return the upload email for the folder.
517         */
518        public BoxUploadEmail getUploadEmail() {
519            return this.uploadEmail;
520        }
521
522        /**
523         * Sets the upload email for the folder.
524         * @param uploadEmail the upload email for the folder.
525         */
526        public void setUploadEmail(BoxUploadEmail uploadEmail) {
527            if (this.uploadEmail == uploadEmail) {
528                return;
529            }
530
531            this.removeChildObject("folder_upload_email");
532            this.uploadEmail = uploadEmail;
533
534            if (uploadEmail == null) {
535                this.addPendingChange("folder_upload_email", null);
536            } else {
537                this.addChildObject("folder_upload_email", uploadEmail);
538            }
539        }
540
541        /**
542         * Gets whether or not the folder has any collaborations.
543         * @return true if the folder has collaborations; otherwise false.
544         */
545        public boolean getHasCollaborations() {
546            return this.hasCollaborations;
547        }
548
549        /**
550         * Gets the sync state of the folder.
551         * @return the sync state of the folder.
552         */
553        public SyncState getSyncState() {
554            return this.syncState;
555        }

557        /**
558         * Sets the sync state of the folder.
559         * @param syncState the sync state of the folder.
560         */
561        public void setSyncState(SyncState syncState) {
562            this.syncState = syncState;
563            this.addPendingChange("sync_state", syncState.toJSONValue());
564        }
565
566        /**
567         * Gets the permissions that the current user has on the folder.
568         * @return the permissions that the current user has on the folder.
569         */
570        public EnumSet<Permission> getPermissions() {
571            return this.permissions;
572        }
573
574        /**
575         * Gets whether or not the non-owners can invite collaborators to the folder.
576         * @return [description]
577         */
578        public boolean getCanNonOwnersInvite() {
579            return this.canNonOwnersInvite;
580        }
581
582        @Override
583        public BoxFolder getResource() {
584            return BoxFolder.this;
585        }
586
587        @Override
588        protected void parseJSONMember(JsonObject.Member member) {
589            super.parseJSONMember(member);
590
591            String memberName = member.getName();
592            JsonValue value = member.getValue();
593            if (memberName.equals("folder_upload_email")) {
594                if (this.uploadEmail == null) {
595                    this.uploadEmail = new BoxUploadEmail(value.asObject());
596                } else {
597                    this.uploadEmail.update(value.asObject());
598                }
599
600            } else if (memberName.equals("has_collaborations")) {
601                this.hasCollaborations = value.asBoolean();
602
603            } else if (memberName.equals("sync_state")) {
604                this.syncState = SyncState.fromJSONValue(value.asString());
605
606            } else if (memberName.equals("permissions")) {
607                this.permissions = this.parsePermissions(value.asObject());
608
609            } else if (memberName.equals("can_non_owners_invite")) {
610                this.canNonOwnersInvite = value.asBoolean();
611            }
612        }
613
614        private EnumSet<Permission> parsePermissions(JsonObject jsonObject) {
615            EnumSet<Permission> permissions = EnumSet.noneOf(Permission.class);
616            for (JsonObject.Member member : jsonObject) {
617                JsonValue value = member.getValue();
618                if (value.isNull() || !value.asBoolean()) {
619                    continue;
620                }
621
622                String memberName = member.getName();
623                if (memberName.equals("can_download")) {
624                    permissions.add(Permission.CAN_DOWNLOAD);
625                } else if (memberName.equals("can_upload")) {
626                    permissions.add(Permission.CAN_UPLOAD);
627                } else if (memberName.equals("can_rename")) {
628                    permissions.add(Permission.CAN_RENAME);
629                } else if (memberName.equals("can_delete")) {
630                    permissions.add(Permission.CAN_DELETE);
631                } else if (memberName.equals("can_share")) {
632                    permissions.add(Permission.CAN_SHARE);
633                } else if (memberName.equals("can_invite_collaborator")) {
634                    permissions.add(Permission.CAN_INVITE_COLLABORATOR);
635                } else if (memberName.equals("can_set_share_access")) {
636                    permissions.add(Permission.CAN_SET_SHARE_ACCESS);
637                }
638            }
639
640            return permissions;
641        }
642    }
