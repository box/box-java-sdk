013/**
014 * Represents a set of Box users.
015 *
016 * <p>Unless otherwise noted, the methods in this class can throw an unchecked {@link BoxAPIException} (unchecked

021    private static final URLTemplate GROUPS_URL_TEMPLATE = new URLTemplate("groups");
022    private static final URLTemplate GROUP_URL_TEMPLATE = new URLTemplate("groups/%s");
023    private static final URLTemplate MEMBERSHIPS_URL_TEMPLATE = new URLTemplate("groups/%s/memberships");
024    private static final URLTemplate ADD_MEMBERSHIP_URL_TEMPLATE = new URLTemplate("group_memberships");
025    private static final URLTemplate COLLABORATIONS_URL_TEMPLATE = new URLTemplate("groups/%s/collaborations");

027    /**

033        super(api, id);

036    /**
037     * Creates a new group with a specified name.
038     * @param  api  the API connection to be used by the group.
039     * @param  name the name of the new group.
040     * @return      info about the created group.
041     */
042    public static BoxGroup.Info createGroup(BoxAPIConnection api, String name) {
043        return createGroup(api, name, null, null, null, null, null);
044    }

046    /**
047     * Creates a new group with a specified name.
048     * @param  api  the API connection to be used by the group.
049     * @param  name the name of the new group.
050     * @param provenance the provenance of the new group
051     * @param externalSyncIdentifier the external_sync_identifier of the new group
052     * @param description the description of the new group
053     * @param invitabilityLevel the invitibility_level of the new group
054     * @param memberViewabilityLevel the member_viewability_level of the new group
055     * @return      info about the created group.
056     */
057    public static BoxGroup.Info createGroup(BoxAPIConnection api, String name, String provenance,
058                                            String externalSyncIdentifier, String description,
059                                            String invitabilityLevel, String memberViewabilityLevel) {
060        JsonObject requestJSON = new JsonObject();
061        requestJSON.add("name", name);
062
063        if (provenance != null) {
064            requestJSON.add("provenance", provenance);
065        }
066        if (externalSyncIdentifier != null) {
067            requestJSON.add("external_sync_identifier", externalSyncIdentifier);
068        }
069        if (description != null) {
070            requestJSON.add("description", description);
071        }
072        if (invitabilityLevel != null) {
073            requestJSON.add("invitability_level", invitabilityLevel);
074        }
075        if (memberViewabilityLevel != null) {
076            requestJSON.add("member_viewability_level", memberViewabilityLevel);
077        }
078
079        URL url = GROUPS_URL_TEMPLATE.build(api.getBaseURL());
080        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
081        request.setBody(requestJSON.toString());
082        BoxJSONResponse response = (BoxJSONResponse) request.send();
083        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
084
085        BoxGroup group = new BoxGroup(api, responseJSON.get("id").asString());
086        return group.new Info(responseJSON);
087    }

095        return new Iterable<BoxGroup.Info>() {
096            public Iterator<BoxGroup.Info> iterator() {
097                URL url = GROUPS_URL_TEMPLATE.build(api.getBaseURL());
098                return new BoxGroupIterator(api, url);
099            }
100        };
101    }

103    /**
104     * Gets information about this group.
105     * @return info about this group.
106     */
107    public Info getInfo() {
108        URL url = GROUP_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
109        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
110        BoxJSONResponse response = (BoxJSONResponse) request.send();
111        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
112        return new Info(responseJSON);
113    }

115    /**
116     * Gets information about all of the group memberships for this group.
117     * @return a collection of information about the group memberships for this group.
118     */
119    public Collection<BoxGroupMembership.Info> getMemberships() {
120        BoxAPIConnection api = this.getAPI();
121        URL url = MEMBERSHIPS_URL_TEMPLATE.build(api.getBaseURL(), this.getID());
122
123        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
124        BoxJSONResponse response = (BoxJSONResponse) request.send();
125        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
126
127        int entriesCount = responseJSON.get("total_count").asInt();
128        Collection<BoxGroupMembership.Info> memberships = new ArrayList<BoxGroupMembership.Info>(entriesCount);
129        JsonArray entries = responseJSON.get("entries").asArray();
130        for (JsonValue entry : entries) {
131            JsonObject entryObject = entry.asObject();
132            BoxGroupMembership membership = new BoxGroupMembership(api, entryObject.get("id").asString());
133            BoxGroupMembership.Info info = membership.new Info(entryObject);
134            memberships.add(info);
135        }
136
137        return memberships;
138    }

140    /**
141     * Adds a member to this group with the default role.
142     * @param  user the member to be added to this group.
143     * @return      info about the new group membership.
144     */
145    public BoxGroupMembership.Info addMembership(BoxUser user) {
146        return this.addMembership(user, null);
147    }
148
149    /**
150     * Adds a member to this group with the specified role.
151     * @param  user the member to be added to this group.
152     * @param  role the role of the user in this group. Can be null to assign the default role.
153     * @return      info about the new group membership.
154     */
155    public BoxGroupMembership.Info addMembership(BoxUser user, Role role) {
156        BoxAPIConnection api = this.getAPI();
157
158        JsonObject requestJSON = new JsonObject();
159        requestJSON.add("user", new JsonObject().add("id", user.getID()));
160        requestJSON.add("group", new JsonObject().add("id", this.getID()));
161        if (role != null) {
162            requestJSON.add("role", role.toJSONString());
163        }
164
165        URL url = ADD_MEMBERSHIP_URL_TEMPLATE.build(api.getBaseURL());
166        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
167        request.setBody(requestJSON.toString());
168        BoxJSONResponse response = (BoxJSONResponse) request.send();
169        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
170
171        BoxGroupMembership membership = new BoxGroupMembership(api, responseJSON.get("id").asString());
172        return membership.new Info(responseJSON);
173    }

175    /**
176     * Gets information about all of the collaborations for this group.
177     * @return a collection of information about the collaborations for this group.
178     */
179    public Collection<BoxCollaboration.Info> getCollaborations() {
180        BoxAPIConnection api = this.getAPI();
181        URL url = COLLABORATIONS_URL_TEMPLATE.build(api.getBaseURL(), this.getID());
182
183        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
184        BoxJSONResponse response = (BoxJSONResponse) request.send();
185        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
186
187        int entriesCount = responseJSON.get("total_count").asInt();
188        Collection<BoxCollaboration.Info> collaborations = new ArrayList<BoxCollaboration.Info>(entriesCount);
189        JsonArray entries = responseJSON.get("entries").asArray();
190        for (JsonValue entry : entries) {
191            JsonObject entryObject = entry.asObject();
192            BoxCollaboration collaboration = new BoxCollaboration(api, entryObject.get("id").asString());
193            BoxCollaboration.Info info = collaboration.new Info(entryObject);
194            collaborations.add(info);
195        }
196
197        return collaborations;
198    }

200    /**
201     * Deletes this group.
202     */
203    public void delete() {
204        URL url = GROUP_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
205        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
206        BoxAPIResponse response = request.send();
207        response.disconnect();
208    }

251            super.parseJSONMember(member);

256                this.description = value.asString();
257            } else if (memberName.equals("external_sync_identifier")) {
258                this.externalSyncIdentifier = value.asString();
259            } else if (memberName.equals("invitability_level")) {
260                this.invitabilityLevel = value.asString();
261            } else if (memberName.equals("member_viewability_level")) {
262                this.memberViewabilityLevel = value.asString();
263            } else if (memberName.equals("provenance")) {
264                this.provenance = value.asString();
265            }
266        }
