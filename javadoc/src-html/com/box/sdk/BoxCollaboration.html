001package com.box.sdk;
002
003import java.net.URL;
004import java.text.ParseException;
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.Date;
008
009import com.eclipsesource.json.JsonArray;
010import com.eclipsesource.json.JsonObject;
011import com.eclipsesource.json.JsonValue;
012
013public class BoxCollaboration extends BoxResource {
014    private static final URLTemplate COLLABORATIONS_URL_TEMPLATE = new URLTemplate("collaborations");
015    private static final URLTemplate PENDING_COLLABORATIONS_URL = new URLTemplate("collaborations?status=pending");
016    private static final URLTemplate COLLABORATION_URL_TEMPLATE = new URLTemplate("collaborations/%s");
017
018    public BoxCollaboration(BoxAPIConnection api, String id) {
019        super(api, id);
020    }
021
022    public static Collection<Info> getPendingCollaborations(BoxAPIConnection api) {
023        URL url = PENDING_COLLABORATIONS_URL.build(api.getBaseURL());
024
025        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
026        BoxJSONResponse response = (BoxJSONResponse) request.send();
027        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
028
029        int entriesCount = responseJSON.get("total_count").asInt();
030        Collection<BoxCollaboration.Info> collaborations = new ArrayList<BoxCollaboration.Info>(entriesCount);
031        JsonArray entries = responseJSON.get("entries").asArray();
032        for (JsonValue entry : entries) {
033            JsonObject entryObject = entry.asObject();
034            BoxCollaboration collaboration = new BoxCollaboration(api, entryObject.get("id").asString());
035            BoxCollaboration.Info info = collaboration.new Info(entryObject);
036            collaborations.add(info);
037        }
038
039        return collaborations;
040    }
041
042    public Info getInfo() {
043        BoxAPIConnection api = this.getAPI();
044        URL url = COLLABORATIONS_URL_TEMPLATE.build(api.getBaseURL());
045
046        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
047        BoxJSONResponse response = (BoxJSONResponse) request.send();
048        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
049        return new Info(jsonObject);
050    }
051
052    public void updateInfo(Info info) {
053        BoxAPIConnection api = this.getAPI();
054        URL url = COLLABORATION_URL_TEMPLATE.build(api.getBaseURL(), this.getID());
055
056        BoxJSONRequest request = new BoxJSONRequest(api, url, "PUT");
057        request.setBody(info.getPendingChanges());
058        BoxJSONResponse response = (BoxJSONResponse) request.send();
059        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
060        info.update(jsonObject);
061    }
062
063    public void delete() {
064        BoxAPIConnection api = this.getAPI();
065        URL url = COLLABORATION_URL_TEMPLATE.build(api.getBaseURL(), this.getID());
066
067        BoxAPIRequest request = new BoxAPIRequest(api, url, "DELETE");
068        BoxAPIResponse response = request.send();
069        response.disconnect();
070    }
071
072    public class Info extends BoxResource.Info {
073        private BoxUser.Info createdBy;
074        private Date createdAt;
075        private Date modifiedAt;
076        private Date expiresAt;
077        private Status status;
078        private BoxCollaborator.Info accessibleBy;
079        private Role role;
080        private Date acknowledgedAt;
081        private BoxFolder.Info item;
082
083        Info(JsonObject jsonObject) {
084            super(jsonObject);
085        }
086
087        public BoxUser.Info getCreatedBy() {
088            return this.createdBy;
089        }
090
091        public Date getCreatedAt() {
092            return this.createdAt;
093        }
094
095        public Date getModifiedAt() {
096            return this.modifiedAt;
097        }
098
099        public Date getExpiresAt() {
100            return this.expiresAt;
101        }
102
103        public Status getStatus() {
104            return this.status;
105        }
106
107        public void setStatus(Status status) {
108            this.status = status;
109            this.addPendingChange("status", status.name().toLowerCase());
110        }
111
112        public BoxCollaborator.Info getAccessibleBy() {
113            return this.accessibleBy;
114        }
115
116        public Role getRole() {
117            return this.role;
118        }
119
120        public void setRole(Role role) {
121            this.role = role;
122            this.addPendingChange("role", role.toJSONString());
123        }
124
125        public Date getAcknowledgedAt() {
126            return this.acknowledgedAt;
127        }
128
129        public BoxFolder.Info getItem() {
130            return this.item;
131        }
132
133        @Override
134        public BoxCollaboration getResource() {
135            return BoxCollaboration.this;
136        }
137
138        @Override
139        protected void parseJSONMember(JsonObject.Member member) {
140            super.parseJSONMember(member);
141
142            String memberName = member.getName();
143            JsonValue value = member.getValue();
144            try {
145                switch (memberName) {
146                    case "created_by":
147                        JsonObject userJSON = value.asObject();
148                        if (this.createdBy == null) {
149                            String userID = userJSON.get("id").asString();
150                            BoxUser user = new BoxUser(getAPI(), userID);
151                            this.createdBy = user.new Info(userJSON);
152                        } else {
153                            this.createdBy.update(userJSON);
154                        }
155                        break;
156                    case "created_at":
157                        this.createdAt = BoxDateParser.parse(value.asString());
158                        break;
159                    case "modified_at":
160                        this.modifiedAt = BoxDateParser.parse(value.asString());
161                        break;
162                    case "expires_at":
163                        this.expiresAt = BoxDateParser.parse(value.asString());
164                        break;
165                    case "status":
166                        String statusString = value.asString().toUpperCase();
167                        this.status = Status.valueOf(statusString);
168                        break;
169                    case "accessible_by":
170                        userJSON = value.asObject();
171                        if (this.accessibleBy == null) {
172                            String userID = userJSON.get("id").asString();
173                            BoxUser user = new BoxUser(getAPI(), userID);
174                            BoxUser.Info userInfo = user.new Info(userJSON);
175                            this.accessibleBy = userInfo;
176                        } else {
177                            this.accessibleBy.update(userJSON);
178                        }
179                        break;
180                    case "role":
181                        this.role = Role.fromJSONString(value.asString());
182                        break;
183                    case "acknowledged_at":
184                        this.acknowledgedAt = BoxDateParser.parse(value.asString());
185                        break;
186                    case "item":
187                        JsonObject folderJSON = value.asObject();
188                        if (this.item == null) {
189                            String folderID = folderJSON.get("id").asString();
190                            BoxFolder folder = new BoxFolder(getAPI(), folderID);
191                            this.item = folder.new Info(folderJSON);
192                        } else {
193                            this.item.update(folderJSON);
194                        }
195                        break;
196                    default:
197                        break;
198                }
199            } catch (ParseException e) {
200                assert false : "A ParseException indicates a bug in the SDK.";
201            }
202        }
203    }
204
205    public enum Status {
206        ACCEPTED,
207        PENDING,
208        REJECTED;
209    }
210
211    public enum Role {
212        EDITOR ("editor"),
213        VIEWER ("viewer"),
214        PREVIEWER ("previewer"),
215        UPLOADER ("uploader"),
216        PREVIEWER_UPLOADER ("previewer uploader"),
217        VIEWER_UPLOADER ("viewer uploader"),
218        CO_OWNER ("co-owner"),
219        OWNER ("owner");
220
221        private final String jsonValue;
222
223        private Role(String jsonValue) {
224            this.jsonValue = jsonValue;
225        }
226
227        public static Role fromJSONString(String jsonValue) {
228            switch (jsonValue) {
229                case "editor":
230                    return EDITOR;
231                case "viewer":
232                    return VIEWER;
233                case "previewer":
234                    return PREVIEWER;
235                case "uploader":
236                    return UPLOADER;
237                case "previewer uploader":
238                    return PREVIEWER_UPLOADER;
239                case "viewer uploader":
240                    return VIEWER_UPLOADER;
241                case "co-owner":
242                    return CO_OWNER;
243                case "owner":
244                    return OWNER;
245                default:
246                    throw new IllegalArgumentException("The provided JSON value isn't a valid Role.");
247            }
248        }
249
250        public String toJSONString() {
251            return this.jsonValue;
252        }
253    }
