001package com.box.sdk;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URL;
007import java.text.ParseException;
008import java.util.Date;
009
010import com.eclipsesource.json.JsonObject;
011import com.eclipsesource.json.JsonValue;
012
013public class BoxFileVersion extends BoxResource {
014    private static final URLTemplate CONTENT_URL_TEMPLATE = new URLTemplate("files/%s/content?version=%s");
015    private static final URLTemplate VERSION_URL_TEMPLATE = new URLTemplate("files/%s/versions/%s");
016    private static final int BUFFER_SIZE = 8192;
017
018    private final String fileID;
019
020    private String sha1;
021    private String name;
022    private long size;
023    private Date createdAt;
024    private Date modifiedAt;
025    private BoxUser.Info modifiedBy;
026
027    BoxFileVersion(BoxAPIConnection api, JsonObject jsonObject, String fileID) {
028        super(api, jsonObject.get("id").asString());
029
030        this.fileID = fileID;
031        for (JsonObject.Member member : jsonObject) {
032            JsonValue value = member.getValue();
033            if (value.isNull()) {
034                continue;
035            }
036
037            try {
038                switch (member.getName()) {
039                    case "sha1":
040                        this.sha1 = value.asString();
041                        break;
042                    case "name":
043                        this.name = value.asString();
044                        break;
045                    case "size":
046                        this.size = Double.valueOf(value.toString()).longValue();
047                        break;
048                    case "created_at":
049                        this.createdAt = BoxDateParser.parse(value.asString());
050                        break;
051                    case "modified_at":
052                        this.modifiedAt = BoxDateParser.parse(value.asString());
053                        break;
054                    case "modified_by":
055                        JsonObject userJSON = value.asObject();
056                        String userID = userJSON.get("id").asString();
057                        BoxUser user = new BoxUser(getAPI(), userID);
058                        this.modifiedBy = user.new Info(userJSON);
059                        break;
060                    default:
061                        break;
062                }
063            } catch (ParseException e) {
064                assert false : "A ParseException indicates a bug in the SDK.";
065            }
066        }
067    }
068
069    public String getSha1() {
070        return this.sha1;
071    }
072
073    public String getName() {
074        return this.name;
075    }
076
077    public long getSize() {
078        return this.size;
079    }
080
081    public Date getCreatedAt() {
082        return this.createdAt;
083    }
084
085    public Date getModifiedAt() {
086        return this.modifiedAt;
087    }
088
089    public BoxUser.Info getModifiedBy() {
090        return this.modifiedBy;
091    }
092
093    public void delete() {
094        URL url = VERSION_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, this.getID());
095        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
096        BoxAPIResponse response = request.send();
097        response.disconnect();
098    }
099
100    public void download(OutputStream output) {
101        this.download(output, null);
102    }
103
104    public void download(OutputStream output, ProgressListener listener) {
105        URL url = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, this.getID());
106        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
107        BoxAPIResponse response = request.send();
108        InputStream input = response.getBody();
109
110        long totalRead = 0;
111        byte[] buffer = new byte[BUFFER_SIZE];
112        try {
113            int n = input.read(buffer);
114            totalRead += n;
115            while (n != -1) {
116                output.write(buffer, 0, n);
117                if (listener != null) {
118                    listener.onProgressChanged(totalRead, response.getContentLength());
119                }
120                n = input.read(buffer);
121                totalRead += n;
122            }
123        } catch (IOException e) {
124            throw new BoxAPIException("Couldn't connect to the Box API due to a network error.", e);
125        }
126
127        response.disconnect();
128    }
129
130    public void promote() {
131        URL url = VERSION_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, "current");
132
133        JsonObject jsonObject = new JsonObject();
134        jsonObject.add("type", "file_version");
135        jsonObject.add("id", this.getID());
136
137        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
138        request.setBody(jsonObject.toString());
139        BoxAPIResponse response = request.send();
140        response.disconnect();
141    }
142}
