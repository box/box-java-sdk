001package com.box.sdk;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URL;
007import java.text.ParseException;
008import java.util.Date;
009
010import com.eclipsesource.json.JsonObject;
011import com.eclipsesource.json.JsonValue;
012
013/**
014 * Represents a particular version of a file on Box.
015 */
016public class BoxFileVersion extends BoxResource {
017    private static final URLTemplate CONTENT_URL_TEMPLATE = new URLTemplate("files/%s/content?version=%s");
018    private static final URLTemplate VERSION_URL_TEMPLATE = new URLTemplate("files/%s/versions/%s");
019    private static final int BUFFER_SIZE = 8192;
020
021    private final String fileID;
022
023    private String versionID;
024    private String sha1;
025    private String name;
026    private long size;
027    private Date createdAt;
028    private Date modifiedAt;
029    private BoxUser.Info modifiedBy;
030    private Date trashedAt;
031
032    /**
033     * Constructs a BoxFileVersion from a JSON string.
034     * @param  api    the API connection to be used by the file.
035     * @param  json   the JSON encoded file version.
036     * @param  fileID the ID of the file.
037     */
038    public BoxFileVersion(BoxAPIConnection api, String json, String fileID) {
039        this(api, JsonObject.readFrom(json), fileID);
040    }
041
042    BoxFileVersion(BoxAPIConnection api, JsonObject jsonObject, String fileID) {
043        super(api, jsonObject.get("id").asString());
044
045        this.fileID = fileID;
046        for (JsonObject.Member member : jsonObject) {
047            JsonValue value = member.getValue();
048            if (value.isNull()) {
049                continue;
050            }
051
052            try {
053                String memberName = member.getName();
054                if (memberName.equals("id")) {
055                    this.versionID = value.asString();
056                } else if (memberName.equals("sha1")) {
057                    this.sha1 = value.asString();
058                } else if (memberName.equals("name")) {
059                    this.name = value.asString();
060                } else if (memberName.equals("size")) {
061                    this.size = Double.valueOf(value.toString()).longValue();
062                } else if (memberName.equals("created_at")) {
063                    this.createdAt = BoxDateFormat.parse(value.asString());
064                } else if (memberName.equals("modified_at")) {
065                    this.modifiedAt = BoxDateFormat.parse(value.asString());
066                } else if (memberName.equals("trashed_at")) {
067                    this.trashedAt = BoxDateFormat.parse(value.asString());
068                } else if (memberName.equals("modified_by")) {
069                    JsonObject userJSON = value.asObject();
070                    String userID = userJSON.get("id").asString();
071                    BoxUser user = new BoxUser(getAPI(), userID);
072                    this.modifiedBy = user.new Info(userJSON);
073                }
074            } catch (ParseException e) {
075                assert false : "A ParseException indicates a bug in the SDK.";
076            }
077        }
078    }
079
080    /**
081     * Gets the version ID of this version of the file.
082     * @return the version ID of this version of the file.
083     */
084    public String getVersionID() {
085        return this.versionID;
086    }
087
088    /**
089     * Gets the SHA1 hash of this version of the file.
090     * @return the SHA1 hash of this version of the file.
091     */
092    public String getSha1() {
093        return this.sha1;
094    }
095
096    /**
097     * Gets the name of this version of the file.
098     * @return the name of this version of the file.
099     */
100    public String getName() {
101        return this.name;
102    }
103
104    /**
105     * Gets the size of this version of the file.
106     * @return the size of this version of the file.
107     */
108    public long getSize() {
109        return this.size;
110    }
111
112    /**
113     * Gets the time that this version of the file was created.
114     * @return the time that this version of the file was created.
115     */
116    public Date getCreatedAt() {
117        return this.createdAt;
118    }
119
120    /**
121     * Gets the time that this version of the file was modified.
122     * @return the time that this version of the file was modified.
123     */
124    public Date getModifiedAt() {
125        return this.modifiedAt;
126    }
127
128    /**
129     * Gets the time that this version of the file was deleted.
130     * @return the time that this version of the file was deleted.
131     */
132    public Date getTrashedAt() {
133        return this.trashedAt;
134    }
135
136    /**
137     * Gets information about the user who last modified this version of the file.
138     * @return info about the user who last modified this version of the file.
139     */
140    public BoxUser.Info getModifiedBy() {
141        return this.modifiedBy;
142    }
143
144    /**
145     * Deletes this version of the file.
146     */
147    public void delete() {
148        URL url = VERSION_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, this.getID());
149        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
150        BoxAPIResponse response = request.send();
151        response.disconnect();
152    }
153
154    /**
155     * Downloads this version of the file to a given OutputStream.
156     * @param output the stream to where the file will be written.
157     */
158    public void download(OutputStream output) {
159        this.download(output, null);
160    }
161
162    /**
163     * Downloads this version of the file to a given OutputStream while reporting the progress to a ProgressListener.
164     * @param output   the stream to where the file will be written.
165     * @param listener a listener for monitoring the download's progress.
166     */
167    public void download(OutputStream output, ProgressListener listener) {
168        URL url = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, this.getID());
169        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
170        BoxAPIResponse response = request.send();
171        InputStream input = response.getBody(listener);
172
173        long totalRead = 0;
174        byte[] buffer = new byte[BUFFER_SIZE];
175        try {
176            int n = input.read(buffer);
177            totalRead += n;
178            while (n != -1) {
179                output.write(buffer, 0, n);
180                n = input.read(buffer);
181                totalRead += n;
182            }
183        } catch (IOException e) {
184            throw new BoxAPIException("Couldn't connect to the Box API due to a network error.", e);
185        }
186
187        response.disconnect();
188    }
189
190    /**
191     * Promotes this version of the file to be the latest version.
192     */
193    public void promote() {
194        URL url = VERSION_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.fileID, "current");
195
196        JsonObject jsonObject = new JsonObject();
197        jsonObject.add("type", "file_version");
198        jsonObject.add("id", this.getID());
199
200        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
201        request.setBody(jsonObject.toString());
202        BoxAPIResponse response = request.send();
203        response.disconnect();
204    }
205}
