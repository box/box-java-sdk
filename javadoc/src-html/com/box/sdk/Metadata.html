001package com.box.sdk;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import com.eclipsesource.json.JsonArray;
007import com.eclipsesource.json.JsonObject;
008import com.eclipsesource.json.JsonValue;
009
010/**
011 * The Metadata class represents one type instance of Box metadata.
012 *
013 * Learn more about Box metadata:
014 * https://developers.box.com/metadata-api/
015 */
016public class Metadata {
017    private final JsonObject values;
018    private JsonArray operations;
019
020    /**
021     * Creates an empty metadata.
022     */
023    public Metadata() {
024        this.values = new JsonObject();
025    }
026
027    /**
028     * Creates a new metadata.
029     * @param values the initial metadata values.
030     */
031    Metadata(JsonObject values) {
032        this.values = values;
033    }
034
035    /**
036     * Creates a copy of another metadata.
037     * @param other the other metadata object to copy.
038     */
039    public Metadata(Metadata other) {
040        this.values = new JsonObject(other.values);
041    }
042
043    /**
044     * Returns the 36 character UUID to identify the metadata object.
045     * @return the metadata ID.
046     */
047    public String getID() {
048        return this.get("/$id");
049    }
050
051    /**
052     * Returns the metadata type.
053     * @return the metadata type.
054     */
055    public String getTypeName() {
056        return this.get("/$type");
057    }
058
059    /**
060     * Returns the parent object ID (typically the file ID).
061     * @return the parent object ID.
062     */
063    public String getParentID() {
064        return this.get("/$parent");
065    }
066
067    /**
068     * Returns the scope.
069     * @return the scope.
070     */
071    public String getScope() {
072        return this.get("/$scope");
073    }
074
075    /**
076     * Returns the template name.
077     * @return the template name.
078     */
079    public String getTemplateName() {
080        return this.get("/$template");
081    }
082
083    /**
084     * Adds a new metadata value.
085     * @param path the path that designates the key. Must be prefixed with a "/".
086     * @param value the value.
087     * @return this metadata object.
088     */
089    public Metadata add(String path, String value) {
090        this.values.add(this.pathToProperty(path), value);
091        this.addOp("add", path, value);
092        return this;
093    }
094
095    /**
096     * Replaces an existing metadata value.
097     * @param path the path that designates the key. Must be prefixed with a "/".
098     * @param value the value.
099     * @return this metadata object.
100     */
101    public Metadata replace(String path, String value) {
102        this.values.set(this.pathToProperty(path), value);
103        this.addOp("replace", path, value);
104        return this;
105    }
106
107    /**
108     * Removes an existing metadata value.
109     * @param path the path that designates the key. Must be prefixed with a "/".
110     * @return this metadata object.
111     */
112    public Metadata remove(String path) {
113        this.values.remove(this.pathToProperty(path));
114        this.addOp("remove", path, null);
115        return this;
116    }
117
118    /**
119     * Tests that a property has the expected value.
120     * @param path the path that designates the key. Must be prefixed with a "/".
121     * @param value the expected value.
122     * @return this metadata object.
123     */
124    public Metadata test(String path, String value) {
125        this.addOp("test", path, value);
126        return this;
127    }
128
129    /**
130     * Returns a value.
131     * @param path the path that designates the key. Must be prefixed with a "/".
132     * @return the metadata property value.
133     */
134    public String get(String path) {
135        final JsonValue value = this.values.get(this.pathToProperty(path));
136        if (value == null) {
137            return null;
138        }
139        if (value.isNumber()) {
140            return value.toString();
141        }
142        return value.asString();
143    }
144
145    /**
146     * Returns a list of metadata property paths.
147     * @return the list of metdata property paths.
148     */
149    public List<String> getPropertyPaths() {
150        List<String> result = new ArrayList<String>();
151
152        for (String property : this.values.names()) {
153            if (!property.startsWith("$")) {
154                result.add(this.propertyToPath(property));
155            }
156        }
157
158        return result;
159    }
160
161    /**
162     * Returns the JSON patch string with all operations.
163     * @return the JSON patch string.
164     */
165    public String getPatch() {
166        if (this.operations == null) {
167            return "[]";
168        }
169        return this.operations.toString();
170    }
171
172    /**
173     * Returns the JSON representation of this metadata.
174     * @return the JSON representation of this metadata.
175     */
176    @Override
177    public String toString() {
178        return this.values.toString();
179    }
180
181    /**
182     * Converts a JSON patch path to a JSON property name.
183     * Currently the metadata API only supports flat maps.
184     * @param path the path that designates the key.  Must be prefixed with a "/".
185     * @return the JSON property name.
186     */
187    private String pathToProperty(String path) {
188        if (path == null || !path.startsWith("/")) {
189            throw new IllegalArgumentException("Path must be prefixed with a \"/\".");
190        }
191        return path.substring(1);
192    }
193
194    /**
195     * Converts a JSON property name to a JSON patch path.
196     * @param property the JSON property name.
197     * @return the path that designates the key.
198     */
199    private String propertyToPath(String property) {
200        if (property == null) {
201            throw new IllegalArgumentException("Property must not be null.");
202        }
203        return "/" + property;
204    }
205
206    /**
207     * Adds a patch operation.
208     * @param op the operation type. Must be add, replace, remove, or test.
209     * @param path the path that designates the key. Must be prefixed with a "/".
210     * @param value the value to be set.
211     */
212    private void addOp(String op, String path, String value) {
213        if (this.operations == null) {
214            this.operations = new JsonArray();
215        }
216
217        this.operations.add(new JsonObject()
218                .add("op", op)
219                .add("path", path)
220                .add("value", value));
221    }
222}
