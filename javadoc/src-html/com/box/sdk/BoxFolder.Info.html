014/**
015 * Represents a folder on Box. This class can be used to iterate through a folder's contents, collaborate a folder with
016 * another user or group, and perform other common folder operations (move, copy, delete, etc.).
017 */
018public final class BoxFolder extends BoxItem implements Iterable<BoxItem.Info> {
019    private static final String UPLOAD_FILE_URL_BASE = "https://upload.box.com/api/2.0/";
020    private static final URLTemplate CREATE_FOLDER_URL = new URLTemplate("folders");
021    private static final URLTemplate COPY_FOLDER_URL = new URLTemplate("folders/%s/copy");
022    private static final URLTemplate DELETE_FOLDER_URL = new URLTemplate("folders/%s?recursive=%b");
023    private static final URLTemplate FOLDER_INFO_URL_TEMPLATE = new URLTemplate("folders/%s");
024    private static final URLTemplate UPLOAD_FILE_URL = new URLTemplate("files/content");
025    private static final URLTemplate ADD_COLLABORATION_URL = new URLTemplate("collaborations");
026    private static final URLTemplate GET_COLLABORATIONS_URL = new URLTemplate("folders/%s/collaborations");
027
028    private final URL folderURL;
029
030    /**
031     * Constructs a BoxFolder for a folder with a given ID.
032     * @param  api the API connection to be used by the folder.
033     * @param  id  the ID of the folder.
034     */
035    public BoxFolder(BoxAPIConnection api, String id) {
036        super(api, id);
037
038        this.folderURL = FOLDER_INFO_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
039    }

041    /**
042     * Gets the current user's root folder.
043     * @param  api the API connection to be used by the folder.
044     * @return     the user's root folder.
045     */
046    public static BoxFolder getRootFolder(BoxAPIConnection api) {
047        return new BoxFolder(api, "0");
048    }

050    /**
051     * Adds a collaborator to this folder.
052     * @param  collaborator the collaborator to add.
053     * @param  role         the role of the collaborator.
054     * @return              info about the new collaboration.
055     */
056    public BoxCollaboration.Info collaborate(BoxCollaborator collaborator, BoxCollaboration.Role role) {
057        JsonObject accessibleByField = new JsonObject();
058        accessibleByField.add("id", collaborator.getID());
059
060        if (collaborator instanceof BoxUser) {
061            accessibleByField.add("type", "user");
062        } else {
063            throw new IllegalArgumentException("The given collaborator is of an unknown type.");
064        }
065
066        return this.collaborate(accessibleByField, role);
067    }
068
069    /**
070     * Adds a collaborator to this folder. An email will be sent to the collaborator if they don't already have a Box
071     * account.
072     * @param  email the email address of the collaborator to add.
073     * @param  role  the role of the collaborator.
074     * @return       info about the new collaboration.
075     */
076    public BoxCollaboration.Info collaborate(String email, BoxCollaboration.Role role) {
077        JsonObject accessibleByField = new JsonObject();
078        accessibleByField.add("login", email);
079        accessibleByField.add("type", "user");
080
081        return this.collaborate(accessibleByField, role);
082    }
083
084    private BoxCollaboration.Info collaborate(JsonObject accessibleByField, BoxCollaboration.Role role) {
085        BoxAPIConnection api = this.getAPI();
086        URL url = ADD_COLLABORATION_URL.build(api.getBaseURL());
087
088        JsonObject itemField = new JsonObject();
089        itemField.add("id", this.getID());
090        itemField.add("type", "folder");
091
092        JsonObject requestJSON = new JsonObject();
093        requestJSON.add("item", itemField);
094        requestJSON.add("accessible_by", accessibleByField);
095        requestJSON.add("role", role.toJSONString());
096
097        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
098        request.setBody(requestJSON.toString());
099        BoxJSONResponse response = (BoxJSONResponse) request.send();
100        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
101
102        BoxCollaboration newCollaboration = new BoxCollaboration(api, responseJSON.get("id").asString());
103        BoxCollaboration.Info info = newCollaboration.new Info(responseJSON);
104        return info;
105    }

107    @Override
108    public BoxSharedLink createSharedLink(BoxSharedLink.Access access, Date unshareDate,
109        BoxSharedLink.Permissions permissions) {
110
111        BoxSharedLink sharedLink = new BoxSharedLink(access, unshareDate, permissions);
112        Info info = new Info();
113        info.setSharedLink(sharedLink);
114
115        this.updateInfo(info);
116        return info.getSharedLink();
117    }

119    /**
120     * Gets information about all of the collaborations for this folder.
121     * @return a collection of information about the collaborations for this folder.
122     */
123    public Collection<BoxCollaboration.Info> getCollaborations() {
124        BoxAPIConnection api = this.getAPI();
125        URL url = GET_COLLABORATIONS_URL.build(api.getBaseURL(), this.getID());
126
127        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
128        BoxJSONResponse response = (BoxJSONResponse) request.send();
129        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
130
131        int entriesCount = responseJSON.get("total_count").asInt();
132        Collection<BoxCollaboration.Info> collaborations = new ArrayList<BoxCollaboration.Info>(entriesCount);
133        JsonArray entries = responseJSON.get("entries").asArray();
134        for (JsonValue entry : entries) {
135            JsonObject entryObject = entry.asObject();
136            BoxCollaboration collaboration = new BoxCollaboration(api, entryObject.get("id").asString());
137            BoxCollaboration.Info info = collaboration.new Info(entryObject);
138            collaborations.add(info);
139        }
140
141        return collaborations;
142    }

144    @Override
145    public BoxFolder.Info getInfo() {
146        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.folderURL, "GET");
147        BoxJSONResponse response = (BoxJSONResponse) request.send();
148        return new Info(response.getJSON());
149    }

151    @Override
152    public BoxFolder.Info getInfo(String... fields) {
153        String queryString = new QueryStringBuilder().addFieldsParam(fields).toString();
154        URL url = FOLDER_INFO_URL_TEMPLATE.buildWithQuery(this.getAPI().getBaseURL(), queryString, this.getID());
155
156        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
157        BoxJSONResponse response = (BoxJSONResponse) request.send();
158        return new Info(response.getJSON());
159    }

161    /**
162     * Updates the information about this folder with any info fields that have been modified locally.
163     * @param info the updated info.
164     */
165    public void updateInfo(BoxFolder.Info info) {
166        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
167        request.setBody(info.getPendingChanges());
168        BoxJSONResponse response = (BoxJSONResponse) request.send();
169        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
170        info.update(jsonObject);
171    }

173    @Override

175        return this.copy(destination, null);
176    }

178    @Override
179    public BoxFolder.Info copy(BoxFolder destination, String newName) {
180        URL url = COPY_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID());
181        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
182
183        JsonObject parent = new JsonObject();
184        parent.add("id", destination.getID());
185
186        JsonObject copyInfo = new JsonObject();
187        copyInfo.add("parent", parent);
188        if (newName != null) {
189            copyInfo.add("name", newName);
190        }
191
192        request.setBody(copyInfo.toString());
193        BoxJSONResponse response = (BoxJSONResponse) request.send();
194        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
195        BoxFolder copiedFolder = new BoxFolder(this.getAPI(), responseJSON.get("id").asString());
196        return copiedFolder.new Info(responseJSON);
197    }

199    /**
200     * Creates a new child folder inside this folder.
201     * @param  name the new folder's name.
202     * @return      the created folder.
203     */
204    public BoxFolder createFolder(String name) {
205        JsonObject parent = new JsonObject();
206        parent.add("id", this.getID());
207
208        JsonObject newFolder = new JsonObject();
209        newFolder.add("name", name);
210        newFolder.add("parent", parent);
211
212        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), CREATE_FOLDER_URL.build(this.getAPI().getBaseURL()),
213            "POST");
214        request.setBody(newFolder.toString());
215        BoxJSONResponse response = (BoxJSONResponse) request.send();
216        JsonObject createdFolder = JsonObject.readFrom(response.getJSON());
217
218        return new BoxFolder(this.getAPI(), createdFolder.get("id").asString());
219    }

221    /**
222     * Deletes this folder, optionally recursively deleting all of its contents.
223     * @param recursive true to recursively delete this folder's contents; otherwise false.
224     */
225    public void delete(boolean recursive) {
226        URL url = DELETE_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID(), recursive);
227        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
228        BoxAPIResponse response = request.send();

295    /**
296     * Uploads a new file to this folder with custom upload parameters.
297     * @param  uploadParams the custom upload parameters.
298     * @return              the uploaded file.
299     */
300    public BoxFile uploadFile(FileUploadParams uploadParams) {
301        URL uploadURL = UPLOAD_FILE_URL.build(UPLOAD_FILE_URL_BASE);
302        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
303        request.putField("parent_id", getID());
304
305        if (uploadParams.getSize() > 0) {
306            request.setFile(uploadParams.getContent(), uploadParams.getName(), uploadParams.getSize());
307        } else {
308            request.setFile(uploadParams.getContent(), uploadParams.getName());
309        }
310
311        if (uploadParams.getCreated() != null) {
312            request.putField("content_created_at", uploadParams.getCreated());
313        }
314
315        if (uploadParams.getModified() != null) {
316            request.putField("content_modified_at", uploadParams.getModified());
317        }
318
319        BoxJSONResponse response;
320        if (uploadParams.getProgressListener() == null) {
321            response = (BoxJSONResponse) request.send();
322        } else {
323            response = (BoxJSONResponse) request.send(uploadParams.getProgressListener());
324        }
325        JsonObject collection = JsonObject.readFrom(response.getJSON());
326        JsonArray entries = collection.get("entries").asArray();
327        String uploadedFileID = entries.get(0).asObject().get("id").asString();
328
329        return new BoxFile(getAPI(), uploadedFileID);
330    }

340    /**
341     * Contains additional information about a BoxFolder.
342     */
343    public class Info extends BoxItem.Info {
344        private BoxUploadEmail uploadEmail;
345
346        /**
347         * Constructs an empty Info object.
348         */
349        public Info() {
350            super();
351        }
352
353        /**
354         * Constructs an Info object by parsing information from a JSON string.
355         * @param  json the JSON string to parse.
356         */
357        public Info(String json) {
358            super(json);
359        }
360
361        /**
362         * Constructs an Info object using an already parsed JSON object.
363         * @param  jsonObject the parsed JSON object.
364         */
365        protected Info(JsonObject jsonObject) {
366            super(jsonObject);
367        }
368
369        public BoxUploadEmail getUploadEmail() {
370            return this.uploadEmail;
371        }
372
373        public void setUploadEmail(BoxUploadEmail uploadEmail) {
374            if (this.uploadEmail == uploadEmail) {
375                return;
376            }
377
378            this.removeChildObject("folder_upload_email");
379            this.uploadEmail = uploadEmail;
380
381            if (uploadEmail == null) {
382                this.addPendingChange("folder_upload_email", null);
383            } else {
384                this.addChildObject("folder_upload_email", uploadEmail);
385            }
386        }
387
388        @Override
389        public BoxFolder getResource() {
390            return BoxFolder.this;
391        }
392
393        @Override
394        protected void parseJSONMember(JsonObject.Member member) {
395            super.parseJSONMember(member);
396
397            String memberName = member.getName();
398            JsonValue value = member.getValue();
399            switch (memberName) {
400                case "folder_upload_email":
401                    if (this.uploadEmail == null) {
402                        this.uploadEmail = new BoxUploadEmail(value.asObject());
403                    } else {
404                        this.uploadEmail.update(value.asObject());
405                    }
406                    break;
407                default:
408                    break;
409            }
410        }
411    }
