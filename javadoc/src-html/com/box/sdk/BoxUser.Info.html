001package com.box.sdk;
002
003import java.net.URL;
004import java.text.ParseException;
005import java.util.Date;
006
007import com.eclipsesource.json.JsonObject;
008import com.eclipsesource.json.JsonValue;
009
010public class BoxUser extends BoxResource {
011    private static final URLTemplate GET_USER_URL = new URLTemplate("users/%s");
012    private static final URLTemplate GET_ME_URL = new URLTemplate("users/me");
013
014    public BoxUser(BoxAPIConnection api, String id) {
015        super(api, id);
016    }
017
018    public static BoxUser getCurrentUser(BoxAPIConnection api) {
019        URL url = GET_ME_URL.build(api.getBaseURL());
020        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
021        BoxJSONResponse response = (BoxJSONResponse) request.send();
022        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
023        return new BoxUser(api, jsonObject.get("id").asString());
024    }
025
026    public BoxUser.Info getInfo() {
027        URL url = GET_USER_URL.build(this.getAPI().getBaseURL(), this.getID());
028        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
029        BoxJSONResponse response = (BoxJSONResponse) request.send();
030        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
031        return new Info(jsonObject);
032    }
033
034    public enum Role {
035        ADMIN,
036        COADMIN,
037        USER
038    }
039
040    public enum Status {
041        ACTIVE,
042        INACTIVE,
043        CANNOT_DELETE_EDIT,
044        CANNOT_DELETE_EDIT_UPLOAD
045    }
046
047    public class Info extends BoxResource.Info<BoxUser> {
048        private String name;
049        private String login;
050        private Date createdAt;
051        private Date modifiedAt;
052        private Role role;
053        private String language;
054        private String timezone;
055        private long spaceAmount;
056        private long spaceUsed;
057        private long maxUploadSize;
058        private Status status;
059        private String jobTitle;
060        private String phone;
061        private String address;
062        private String avatarURL;
063
064        protected Info(JsonObject jsonObject) {
065            super(jsonObject);
066        }
067
068        @Override
069        public BoxUser getResource() {
070            return BoxUser.this;
071        }
072
073        public String getName() {
074            return this.name;
075        }
076
077        public String getLogin() {
078            return this.login;
079        }
080
081        public Date getCreatedAt() {
082            return this.createdAt;
083        }
084
085        public Date getModifiedAt() {
086            return this.modifiedAt;
087        }
088
089        public Role getRole() {
090            return this.role;
091        }
092
093        public String getLanguage() {
094            return this.language;
095        }
096
097        public String getTimezone() {
098            return this.timezone;
099        }
100
101        public long getSpaceAmount() {
102            return this.spaceAmount;
103        }
104
105        public long getSpaceUsed() {
106            return this.spaceUsed;
107        }
108
109        public long getMaxUploadSize() {
110            return this.maxUploadSize;
111        }
112
113        public Status getStatus() {
114            return this.status;
115        }
116
117        public String getJobTitle() {
118            return this.jobTitle;
119        }
120
121        public String getPhone() {
122            return this.phone;
123        }
124
125        public String getAddress() {
126            return this.address;
127        }
128
129        public String getAvatarURL() {
130            return this.avatarURL;
131        }
132
133        @Override
134        protected void parseJSONMember(JsonObject.Member member) {
135            try {
136                JsonValue value = member.getValue();
137                switch (member.getName()) {
138                    case "name":
139                        this.name = value.asString();
140                        break;
141                    case "login":
142                        this.login = value.asString();
143                        break;
144                    case "created_at":
145                        this.createdAt = BoxDateParser.parse(value.asString());
146                        break;
147                    case "modified_at":
148                        this.modifiedAt = BoxDateParser.parse(value.asString());
149                        break;
150                    case "role":
151                        this.role = this.parseRole(value);
152                        break;
153                    case "language":
154                        this.language = value.asString();
155                        break;
156                    case "timezone":
157                        this.timezone = value.asString();
158                        break;
159                    case "space_amount":
160                        this.spaceAmount = value.asLong();
161                        break;
162                    case "space_used":
163                        this.spaceUsed = value.asLong();
164                        break;
165                    case "max_upload_size":
166                        this.maxUploadSize = value.asLong();
167                        break;
168                    case "status":
169                        this.status = this.parseStatus(value);
170                        break;
171                    case "job_title":
172                        this.jobTitle = value.asString();
173                        break;
174                    case "phone":
175                        this.jobTitle = value.asString();
176                        break;
177                    case "address":
178                        this.address = value.asString();
179                        break;
180                    case "avatar_url":
181                        this.avatarURL = value.asString();
182                        break;
183                    default:
184                        break;
185                }
186            } catch (ParseException e) {
187                assert false : "A ParseException indicates a bug in the SDK.";
188            }
189        }
190
191        private Role parseRole(JsonValue value) {
192            String roleString = value.asString().toUpperCase();
193            return Role.valueOf(roleString);
194        }
195
196        private Status parseStatus(JsonValue value) {
197            String statusString = value.asString().toUpperCase();
198            return Status.valueOf(statusString);
199        }
200    }
201}
