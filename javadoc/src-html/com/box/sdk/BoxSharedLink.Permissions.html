001package com.box.sdk;
002
003import java.text.ParseException;
004import java.util.Date;
005
006import com.eclipsesource.json.JsonObject;
007import com.eclipsesource.json.JsonValue;
008
009public class BoxSharedLink extends BoxJSONObject {
010    private String url;
011    private String downloadUrl;
012    private String vanityUrl;
013    private boolean isPasswordEnabled;
014    private Date unsharedAt;
015    private long downloadCount;
016    private long previewCount;
017    private Access access;
018    private Permissions permissions;
019
020    public enum Access {
021        DEFAULT (null),
022        OPEN ("open"),
023        COMPANY ("company"),
024        COLLABORATORS ("collaborators");
025
026        private final String jsonValue;
027
028        private Access(String jsonValue) {
029            this.jsonValue = jsonValue;
030        }
031
032        @Override
033        public String toString() {
034            return this.jsonValue;
035        }
036    }
037
038    public BoxSharedLink() { }
039
040    public BoxSharedLink(String json) {
041        super(json);
042    }
043
044    BoxSharedLink(JsonObject jsonObject) {
045        super(jsonObject);
046    }
047
048    public String getURL() {
049        return this.url;
050    }
051
052    public String getDownloadURL() {
053        return this.downloadUrl;
054    }
055
056    public String getVanityURL() {
057        return this.vanityUrl;
058    }
059
060    public boolean getIsPasswordEnabled() {
061        return this.isPasswordEnabled;
062    }
063
064    public Date getUnsharedDate() {
065        return this.unsharedAt;
066    }
067
068    public void setUnsharedDate(Date unsharedDate) {
069        this.unsharedAt = unsharedDate;
070        this.addPendingChange("unshared_at", unsharedDate.toString());
071    }
072
073    public long getDownloadCount() {
074        return this.downloadCount;
075    }
076
077    public long getPreviewCount() {
078        return this.previewCount;
079    }
080
081    public Access getAccess() {
082        return this.access;
083    }
084
085    public void setAccess(Access access) {
086        this.access = access;
087        this.addPendingChange("access", access.toString());
088    }
089
090    public Permissions getPermissions() {
091        return this.permissions;
092    }
093
094    public void setPermissions(Permissions permissions) {
095        this.permissions = permissions;
096        this.addPendingChange("permissions", permissions);
097    }
098
099    @Override
100    void parseJSONMember(JsonObject.Member member) {
101        JsonValue value = member.getValue();
102        try {
103            switch (member.getName()) {
104                case "url":
105                    this.url = value.asString();
106                    break;
107                case "download_url":
108                    this.downloadUrl = value.asString();
109                    break;
110                case "vanity_url":
111                    this.vanityUrl = value.asString();
112                    break;
113                case "is_password_enabled":
114                    this.isPasswordEnabled = value.asBoolean();
115                    break;
116                case "unshared_at":
117                    this.unsharedAt = BoxDateParser.parse(value.asString());
118                    break;
119                case "download_count":
120                    this.downloadCount = Double.valueOf(value.toString()).longValue();
121                    break;
122                case "preview_count":
123                    this.previewCount = Double.valueOf(value.toString()).longValue();
124                    break;
125                case "access":
126                    String accessString = value.asString().toUpperCase();
127                    this.access = Access.valueOf(accessString);
128                    break;
129                case "permissions":
130                    if (this.permissions == null) {
131                        this.permissions = new Permissions(value.asObject());
132                    } else {
133                        this.permissions.update(value.asObject());
134                    }
135                    break;
136                default:
137                    break;
138            }
139        } catch (ParseException e) {
140            assert false : "A ParseException indicates a bug in the SDK.";
141        }
142    }
143
144    public static class Permissions extends BoxJSONObject {
145        private boolean canDownload;
146        private boolean canPreview;
147
148        public Permissions() { }
149
150        public Permissions(String json) {
151            super(json);
152        }
153
154        Permissions(JsonObject jsonObject) {
155            super(jsonObject);
156        }
157
158        public boolean getCanDownload() {
159            return this.canDownload;
160        }
161
162        public void setCanDownload(boolean enabled) {
163            this.canDownload = enabled;
164            this.addPendingChange("can_download", enabled);
165        }
166
167        public boolean getCanPreview() {
168            return this.canPreview;
169        }
170
171        public void setCanPreview(boolean enabled) {
172            this.canPreview = enabled;
173            this.addPendingChange("can_preview", enabled);
174        }
175
176        @Override
177        void parseJSONMember(JsonObject.Member member) {
178            JsonValue value = member.getValue();
179            switch (member.getName()) {
180                case "can_download":
181                    this.canDownload = value.asBoolean();
182                    break;
183                case "can_preview":
184                    this.canPreview = value.asBoolean();
185                    break;
186                default:
187                    break;
188            }
189        }
190    }
191}
