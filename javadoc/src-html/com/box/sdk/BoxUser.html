083    /**
084     * Provisions a new user in an enterprise with additional user information.
085     * @param  api    the API connection to be used by the created user.
086     * @param  login  the email address the user will use to login.
087     * @param  name   the name of the user.
088     * @param  params additional user information.
089     * @return        the created user's info.
090     */
091    public static BoxUser.Info createEnterpriseUser(BoxAPIConnection api, String login, String name,
092        CreateUserParams params) {
093
094        JsonObject requestJSON = new JsonObject();
095        requestJSON.add("login", login);
096        requestJSON.add("name", name);
097
098        if (params != null) {
099            if (params.getRole() != null) {
100                requestJSON.add("role", params.getRole().toJSONValue());
101            }
102
103            if (params.getStatus() != null) {
104                requestJSON.add("status", params.getStatus().toJSONValue());
105            }
106
107            requestJSON.add("language", params.getLanguage());
108            requestJSON.add("is_sync_enabled", params.getIsSyncEnabled());
109            requestJSON.add("job_title", params.getJobTitle());
110            requestJSON.add("phone", params.getPhone());
111            requestJSON.add("address", params.getAddress());
112            requestJSON.add("space_amount", params.getSpaceAmount());
113            requestJSON.add("can_see_managed_users", params.getCanSeeManagedUsers());
114            requestJSON.add("timezone", params.getTimezone());
115            requestJSON.add("is_exempt_from_device_limits", params.getIsExemptFromDeviceLimits());
116            requestJSON.add("is_exempt_from_login_verification", params.getIsExemptFromLoginVerification());
117            requestJSON.add("is_platform_access_only", params.getIsPlatformAccessOnly());
118        }
119
120        URL url = USERS_URL_TEMPLATE.build(api.getBaseURL());
121        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
122        request.setBody(requestJSON.toString());
123        BoxJSONResponse response = (BoxJSONResponse) request.send();
124        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
125
126        BoxUser createdUser = new BoxUser(api, responseJSON.get("id").asString());
127        return createdUser.new Info(responseJSON);
128    }

198    /**
199     * Helper method to abstract out the common logic from the various users methods.
200     *
201     * @param  api        the API connection to be used when retrieving the users.
202     * @param filterTerm    The filter term to lookup users by (login for external, login or name for managed)
203     * @param userType      The type of users we want to search with this request.
204     *                      Valid values are 'managed' (enterprise users), 'external' or 'all'
205     * @param fields        the fields to retrieve. Leave this out for the standard fields.
206     * @return
207     */
208    private static Iterable<BoxUser.Info> getUsersInfoForType(final BoxAPIConnection api,
209          final String filterTerm, final String userType, final String... fields) {
210        return new Iterable<BoxUser.Info>() {
211            public Iterator<BoxUser.Info> iterator() {
212                QueryStringBuilder builder = new QueryStringBuilder();
213                if (filterTerm != null) {
214                    builder.appendParam("filter_term", filterTerm);
215                }
216                if (userType != null) {
217                    builder.appendParam("user_type", userType);
218                }
219                if (fields.length > 0) {
220                    builder.appendParam("fields", fields);
221                }
222                URL url = USERS_URL_TEMPLATE.buildWithQuery(api.getBaseURL(), builder.toString());
223                return new BoxUserIterator(api, url);
224            }
225        };
226    }

229     * Gets information about this user.
230     * @param  fields the optional fields to retrieve.
231     * @return        info about this user.
232     */
233    public BoxUser.Info getInfo(String... fields) {
234        URL url;
235        if (fields.length > 0) {
236            String queryString = new QueryStringBuilder().appendParam("fields", fields).toString();
237            url = USER_URL_TEMPLATE.buildWithQuery(this.getAPI().getBaseURL(), queryString, this.getID());
238        } else {
239            url = USER_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
240        }
241        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
242        BoxJSONResponse response = (BoxJSONResponse) request.send();
243        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
244        return new Info(jsonObject);
245    }

247    /**
248     * Gets information about all of the group memberships for this user.
249     *
250     * <p>Note: This method is only available to enterprise admins.</p>
251     *
252     * @return a collection of information about the group memberships for this user.
253     */
254    public Collection<BoxGroupMembership.Info> getMemberships() {
255        BoxAPIConnection api = this.getAPI();
256        URL url = USER_MEMBERSHIPS_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
257
258        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
259        BoxJSONResponse response = (BoxJSONResponse) request.send();
260        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
261
262        int entriesCount = responseJSON.get("total_count").asInt();
263        Collection<BoxGroupMembership.Info> memberships = new ArrayList<BoxGroupMembership.Info>(entriesCount);
264        JsonArray entries = responseJSON.get("entries").asArray();
265        for (JsonValue entry : entries) {
266            JsonObject entryObject = entry.asObject();
267            BoxGroupMembership membership = new BoxGroupMembership(api, entryObject.get("id").asString());
268            BoxGroupMembership.Info info = membership.new Info(entryObject);
269            memberships.add(info);
270        }
271
272        return memberships;
273    }

307     * Gets a collection of all the email aliases for this user.
308     *
309     * <p>Note that the user's primary login email is not included in the collection of email aliases.</p>
310     *
311     * @return a collection of all the email aliases for this user.
312     */
313    public Collection<EmailAlias> getEmailAliases() {
314        URL url = EMAIL_ALIASES_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
315        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
316        BoxJSONResponse response = (BoxJSONResponse) request.send();
317        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
318
319        int totalCount = responseJSON.get("total_count").asInt();
320        Collection<EmailAlias> emailAliases = new ArrayList<EmailAlias>(totalCount);
321        JsonArray entries = responseJSON.get("entries").asArray();
322        for (JsonValue value : entries) {
323            JsonObject emailAliasJSON = value.asObject();
324            emailAliases.add(new EmailAlias(emailAliasJSON));
325        }
326
327        return emailAliases;
328    }

363    /**
364     * Moves all of the owned content from within one user’s folder into a new folder in another user's account.
365     * You can move folders across users as long as the you have administrative permissions and the 'source'
366     * user owns the folders. Per the documentation at the link below, this will move everything from the root
367     * folder, as this is currently the only mode of operation supported.
368     *
369     * See also https://box-content.readme.io/reference#move-folder-into-another-users-folder
370     *
371     * @param sourceUserID the user id of the user whose files will be the source for this operation
372     * @return info for the newly created folder
373     */
374    public BoxFolder.Info moveFolderToUser(String sourceUserID) {
375        // Currently the API only supports moving of the root folder (0), hence the hard coded "0"
376        URL url = MOVE_FOLDER_TO_USER_TEMPLATE.build(this.getAPI().getBaseURL(), sourceUserID, "0");
377        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "PUT");
378        JsonObject idValue = new JsonObject();
379        idValue.add("id", this.getID());
380        JsonObject ownedBy = new JsonObject();
381        ownedBy.add("owned_by", idValue);
382        request.setBody(ownedBy.toString());
383        BoxJSONResponse response = (BoxJSONResponse) request.send();
384        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
385        BoxFolder movedFolder = new BoxFolder(this.getAPI(), responseJSON.get("id").asString());
386        response.disconnect();
387
388        return movedFolder.new Info(responseJSON);
389    }

819        @Override
820        protected void parseJSONMember(JsonObject.Member member) {
821            super.parseJSONMember(member);
822
823            JsonValue value = member.getValue();
824            String memberName = member.getName();
825            if (memberName.equals("login")) {
826                this.login = value.asString();
827            } else if (memberName.equals("role")) {
828                this.role = Role.fromJSONValue(value.asString());
829            } else if (memberName.equals("language")) {
830                this.language = value.asString();
831            } else if (memberName.equals("timezone")) {
832                this.timezone = value.asString();
833            } else if (memberName.equals("space_amount")) {
834                this.spaceAmount = Double.valueOf(value.toString()).longValue();
835            } else if (memberName.equals("space_used")) {
836                this.spaceUsed = Double.valueOf(value.toString()).longValue();
837            } else if (memberName.equals("max_upload_size")) {
838                this.maxUploadSize = Double.valueOf(value.toString()).longValue();
839            } else if (memberName.equals("status")) {
840                this.status = Status.fromJSONValue(value.asString());
841            } else if (memberName.equals("job_title")) {
842                this.jobTitle = value.asString();
843            } else if (memberName.equals("phone")) {
844                this.phone = value.asString();
845            } else if (memberName.equals("address")) {
846                this.address = value.asString();
847            } else if (memberName.equals("avatar_url")) {
848                this.avatarURL = value.asString();
849            } else if (memberName.equals("canSeeManagedUsers")) {
850                this.canSeeManagedUsers = value.asBoolean();
851            } else if (memberName.equals("is_sync_enabled")) {
852                this.isSyncEnabled = value.asBoolean();
853            } else if (memberName.equals("is_external_collab_restricted")) {
854                this.isExternalCollabRestricted = value.asBoolean();
855            } else if (memberName.equals("is_exempt_from_device_limits")) {
856                this.isExemptFromDeviceLimits = value.asBoolean();
857            } else if (memberName.equals("is_exempt_from_login_verification")) {
858                this.isExemptFromLoginVerification = value.asBoolean();
859            } else if (memberName.equals("is_password_reset_required")) {
860                this.isPasswordResetRequired = value.asBoolean();
861            } else if (memberName.equals("is_platform_access_only")) {
862                this.isPlatformAccessOnly = value.asBoolean();
863            } else if (memberName.equals("enterprise")) {
864                JsonObject jsonObject = value.asObject();
865                if (this.enterprise == null) {
866                    this.enterprise = new BoxEnterprise(jsonObject);
867                } else {
868                    this.enterprise.update(jsonObject);
869                }
870            } else if (memberName.equals("my_tags")) {
871                this.myTags = this.parseMyTags(value.asArray());
872            } else if (memberName.equals("hostname")) {
873                this.hostname = value.asString();
874            }
875        }
876
877        private List<String> parseMyTags(JsonArray jsonArray) {
878            List<String> myTags = new ArrayList<String>(jsonArray.size());
879            for (JsonValue value : jsonArray) {
880                myTags.add(value.asString());
881            }
882
883            return myTags;
884        }
