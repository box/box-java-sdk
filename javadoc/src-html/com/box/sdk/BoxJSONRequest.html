001package com.box.sdk;
002
003import java.net.URL;
004
005public class BoxJSONRequest extends BoxAPIRequest {
006    private String json;
007
008    public BoxJSONRequest(BoxAPIConnection api, URL url, String method) {
009        super(api, url, method);
010
011        this.addHeader("Content-Type", "application/json");
012    }
013
014    @Override
015    public void setBody(String body) {
016        super.setBody(body);
017        this.json = body;
018    }
019
020    @Override
021    protected String bodyToString() {
022        return this.json;
023    }
024}
